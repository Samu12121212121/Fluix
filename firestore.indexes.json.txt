s rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── HELPERS ────────────────────────────────────────────────────────────────

    function isAuth() {
      return request.auth != null;
    }

    function uid() {
      return request.auth.uid;
    }

    // Obtiene el doc del usuario de forma segura (devuelve null si no existe)
    function getUserData() {
      return get(/databases/$(database)/documents/usuarios/$(request.auth.uid)).data;
    }

    function perteneceAEmpresa(empresaId) {
      return isAuth()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && getUserData().empresa_id == empresaId;
    }

    function getRol() {
      return isAuth()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        ? getUserData().rol
        : '';
    }

    function esPropietario(empresaId) {
      return perteneceAEmpresa(empresaId) && getRol() == 'propietario';
    }

    function esAdminOPropietario(empresaId) {
      return perteneceAEmpresa(empresaId)
        && (getRol() == 'admin' || getRol() == 'propietario');
    }

    function esStaffOSuperior(empresaId) {
      return perteneceAEmpresa(empresaId)
        && (getRol() == 'staff' || getRol() == 'admin' || getRol() == 'propietario');
    }

    // ── USUARIOS ───────────────────────────────────────────────────────────────

    match /usuarios/{userId} {
      // El propio usuario puede leer/escribir su doc
      allow read, write: if isAuth() && uid() == userId;
      // Admin/propietario puede leer docs de su empresa
      allow read: if isAuth()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && getUserData().empresa_id == resource.data.empresa_id
        && (getUserData().rol == 'admin' || getUserData().rol == 'propietario');
      // Cualquier auth puede crear su propio doc (registro)
      allow create: if isAuth() && uid() == userId;
    }

    // ── EMPRESAS ───────────────────────────────────────────────────────────────

    match /empresas/{empresaId} {
      // Lectura del doc raíz de empresa
      allow read: if perteneceAEmpresa(empresaId);
      // Solo propietario puede escribir el doc raíz
      allow write: if esPropietario(empresaId);
      // Cualquier auth puede crear una empresa (al registrarse)
      allow create: if isAuth();
      // Bootstrap: permite que el propietario escriba su empresa aunque
      // el doc de configuracion/modulos todavía no exista
      allow write: if isAuth()
        && exists(/databases/$(database)/documents/usuarios/$(request.auth.uid))
        && getUserData().empresa_id == empresaId
        && getUserData().rol == 'propietario';

      // CLIENTES
      match /clientes/{clienteId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esAdminOPropietario(empresaId);
      }

      // RESERVAS
      match /reservas/{reservaId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esStaffOSuperior(empresaId);
        allow create: if isAuth(); // clientes externos pueden crear reservas
      }

      // VALORACIONES — lectura pública, escritura pública (para web WordPress)
      match /valoraciones/{valoracionId} {
        allow read: if true;
        allow create: if true;
        allow update, delete: if esAdminOPropietario(empresaId);
      }

      // RESEÑAS — lectura pública (Google Reviews / WordPress)
      match /resenas/{resenaId} {
        allow read: if true;
        allow create: if true;
        allow update, delete: if esAdminOPropietario(empresaId);
      }

      // PEDIDOS — clientes externos pueden crear
      match /pedidos/{pedidoId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esStaffOSuperior(empresaId);
        allow create: if true;
      }

      // PEDIDOS WHATSAPP
      match /pedidos_whatsapp/{pedidoId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esStaffOSuperior(empresaId);
        allow create: if true;
      }

      // PRODUCTOS — catálogo público para web
      match /productos/{productoId} {
        allow read: if true;
        allow write: if esAdminOPropietario(empresaId);
      }

      // FACTURAS
      match /facturas/{facturaId} {
        allow read, write: if esAdminOPropietario(empresaId);
      }

      // TRANSACCIONES
      match /transacciones/{transaccionId} {
        allow read: if esAdminOPropietario(empresaId);
        allow write: if esStaffOSuperior(empresaId);
      }

      // TAREAS
      match /tareas/{tareaId} {
        allow read, write: if esStaffOSuperior(empresaId);
      }

      // NÓMINAS
      match /nominas/{nominaId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esAdminOPropietario(empresaId);
      }

      // EQUIPOS
      match /equipos/{equipoId} {
        allow read: if esStaffOSuperior(empresaId);
        allow write: if esAdminOPropietario(empresaId);
      }

      // SECCIONES WEB — lectura pública para WordPress
      match /secciones_web/{seccionId} {
        allow read: if true;
        allow write: if esAdminOPropietario(empresaId);
      }

      // DISPOSITIVOS FCM — cada usuario escribe su propio token
      match /dispositivos/{dispositivoId} {
        allow read: if esAdminOPropietario(empresaId);
        allow write: if isAuth();
      }

      // SUSCRIPCIÓN
      match /suscripcion/{docId} {
        allow read: if perteneceAEmpresa(empresaId);
        allow read: if true;
        allow write: if true;
      }

      // ESTADÍSTICAS / subcolecciones (histórico diario, etc.) — también públicas
      match /estadisticas/{docId}/{subcol}/{subId} {
        allow read: if true;
        allow write: if true;
      }

      // CACHÉ (estadísticas calculadas)
      match /cache/{docId} {
        allow read, write: if perteneceAEmpresa(empresaId);
      }

      // CONFIGURACIÓN (módulos, widgets, ajustes)
      // IMPORTANTE: configuracion/{docId} cubre configuracion/modulos y configuracion/widgets
      match /configuracion/{docId} {
        allow read: if perteneceAEmpresa(empresaId);
        allow write: if esPropietario(empresaId);
      }

      // ── CONVENIOS COLECTIVOS ──────────────────────────────────────────────
      // Lectura pública para cargar tablas salariales en la app.
      match /convenios/{convenioId} {
        allow read: if true;
        allow write: if false;
      }

      // FALLBACK para cualquier otra subcolección
      match /{subcollection}/{documentId} {
        allow read: if perteneceAEmpresa(empresaId);
        allow write: if esAdminOPropietario(empresaId);
      }
    }

    // ── REGLA POR DEFECTO: denegar todo lo demás ──────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
