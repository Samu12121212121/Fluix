<script>
// ── Límite de reservas por franja (control local como UI extra) ────────────
const MAX_POR_FRANJA = 2;

function getReservas() {
  try { return JSON.parse(localStorage.getItem('dj_reservas') || '{}'); }
  catch(e) { return {}; }
}
function saveReservas(d) { localStorage.setItem('dj_reservas', JSON.stringify(d)); }
function getCount(fecha, hora) {
  if (!fecha || !hora) return 0;
  const r = getReservas();
  return (r[fecha] && r[fecha][hora]) ? r[fecha][hora] : 0;
}
function addReserva(fecha, hora) {
  const r = getReservas();
  if (!r[fecha]) r[fecha] = {};
  r[fecha][hora] = (r[fecha][hora] || 0) + 1;
  saveReservas(r);
}

function updateSlotInfo() {
  const fecha = document.getElementById('fecha').value;
  const hora  = document.getElementById('hora').value;
  const info  = document.getElementById('slotInfo');
  const btn   = document.getElementById('btnReservar');
  if (!fecha || !hora) { info.textContent = ''; info.className = 'slot-info'; btn.disabled = false; return; }
  const count = getCount(fecha, hora);
  if (count >= MAX_POR_FRANJA) {
    info.textContent = '⚠ Esta franja está completa para ese día';
    info.className = 'slot-info full'; btn.disabled = true;
  } else if (count === MAX_POR_FRANJA - 1) {
    info.textContent = '✓ Último hueco disponible';
    info.className = 'slot-info available'; btn.disabled = false;
  } else {
    info.textContent = '✓ Disponible';
    info.className = 'slot-info available'; btn.disabled = false;
  }
}

document.getElementById('fecha').addEventListener('change', updateSlotInfo);
document.getElementById('hora').addEventListener('change', updateSlotInfo);

// ── Alérgenos toggle ───────────────────────────────────────────────────────
document.querySelectorAll('input[name="alergenos"]').forEach(function(r) {
  r.addEventListener('change', function() {
    document.getElementById('alergenosDetail').classList.toggle('visible', this.value === 'si');
  });
});

// ── Envío del formulario ───────────────────────────────────────────────────
document.getElementById('reservaForm').addEventListener('submit', function(e) {
  e.preventDefault();

  const fecha  = document.getElementById('fecha').value;
  const hora   = document.getElementById('hora').value;
  const nombre = document.getElementById('nombre').value.trim();
  if (!fecha || !hora || !nombre) return;

  if (getCount(fecha, hora) >= MAX_POR_FRANJA) {
    const msg = document.getElementById('formMsg');
    msg.textContent = 'Lo sentimos, esta franja ya no tiene disponibilidad.';
    msg.className = 'form-msg error';
    return;
  }

  const formData = {
    nombre:            nombre,
    telefono:          document.getElementById('telefono').value.trim(),
    fecha:             fecha,
    hora:              hora,
    personas:          document.getElementById('personas').value,
    ubicacion:         document.querySelector('input[name="ubicacion"]:checked').value,
    alergenos:         document.querySelector('input[name="alergenos"]:checked').value,
    alergenos_detalle: document.getElementById('alergenosTexto').value.trim(),
    comentarios:       document.getElementById('comentarios').value.trim(),
  };

  // Enviar a Fluix CRM (el widget lo escucha y guarda en Firestore)
  document.dispatchEvent(new CustomEvent('fluix:reserva', { detail: formData }));

  // Contar localmente para controlar huecos
  addReserva(fecha, hora);

  const ubi = formData.ubicacion;
  const msg = document.getElementById('formMsg');
  msg.textContent = '¡Reserva recibida, ' + nombre + '! Te esperamos el ' + fecha + ' a las ' + hora + ' en ' + (ubi === 'terraza' ? 'la terraza' : 'el salón') + '. Nos pondremos en contacto para confirmar.';
  msg.className = 'form-msg success';

  this.reset();
  document.getElementById('alergenosDetail').classList.remove('visible');
  updateSlotInfo();
});

// ── Fecha mínima = hoy ─────────────────────────────────────────────────────
document.getElementById('fecha').min = new Date().toISOString().split('T')[0];
</script>
